### 关键设计决策

1. **纯前端实现**：所有逻辑（价格计算、交易、天数推进）都在前端完成，无后端依赖
2. **状态管理集中化**：用 Zustand store 管理所有游戏状态，组件通过 selector 订阅
3. **引擎与UI分离**：核心逻辑（价格、交易、天数）与 UI 完全分离，便于测试和复用
4. **对话系统模块化**：对话数据单独管理，对话引擎独立实现

### 5.3 价格计算逻辑

```typescript
function calculateNextPrice(
  currentPrice: number,
  day: number,
  assetType: AssetType
): number {
  if (day === 5) {
    // 泡沫破裂，大幅暴跌
    const dropPercent = randomBetween(DAY_5_DROP_RANGE[0], DAY_5_DROP_RANGE[1]);
    return currentPrice * (1 - dropPercent / 100);
  }

  // Day 1-4 正常上涨
  const [min, max] = DAY_INCREASE_RANGES[day - 1];
  const increasePercent = randomBetween(min, max);

  // 随机波动 ±20%
  const volatility = randomBetween(-20, 20);

  // 黑色郁金香特殊逻辑
  let multiplier = 1;
  if (assetType === AssetType.TULIP_BLACK) {
    if (Math.random() < 0.3) {
      multiplier = 2; // 30% 概率涨幅翻倍
    } else if (Math.random() < 0.2) {
      multiplier = 0.5; // 20% 概率突然暴跌
    }
  }

  const newPrice = currentPrice * (1 + increasePercent / 100) * (1 + volatility / 100) * multiplier;

  return Math.round(newPrice);
}
```

## 6. 交易引擎

### 6.1 买入逻辑

```typescript
function buyAsset(
  assetType: AssetType,
  quantity: number,
  prices: Record<AssetType, number>,
  player: PlayerState
): { success: boolean; error?: string; newPlayerState: PlayerState } {
  const price = prices[assetType];
  const totalCost = price * quantity;

  // 检查是否有足够现金
  if (player.cash < totalCost) {
    return { success: false, error: '现金不足', newPlayerState: player };
  }

  // 计算手续费 (1%)
  const fee = Math.round(totalCost * 0.01);
  const totalWithFee = totalCost + fee;

  if (player.cash < totalWithFee) {
    return { success: false, error: '现金不足（含手续费）', newPlayerState: player };
  }

  // 计算滑点（大额交易触发）
  const tradeValue = totalWithFee;
  const portfolioValue = Object.entries(player.portfolio).reduce(
    (sum, [type, qty]) => sum + prices[type as AssetType] * qty,
    0
  );
  const totalWealth = player.cash + portfolioValue;

  let slippage = 0;
  if (tradeValue > totalWealth * 0.3) {
    slippage = randomBetween(-5, 5); // ±5% 滑点
    const actualPrice = price * (1 + slippage / 100);
    // 重新计算
  }

  // 更新玩家状态
  const newPlayerState = {
    ...player,
    cash: player.cash - totalWithFee,
    portfolio: {
      ...player.portfolio,
      [assetType]: (player.portfolio[assetType] || 0) + quantity,
    },
    tradeHistory: [
      ...player.tradeHistory,
      {
        id: uuid(),
        day: gameState.currentDay,
        assetType,
        action: 'buy',

        fee,

  };

  return { success: true, newPlayerState };
```

### 6.2 卖出逻辑

```typescript
function sellAsset(
  assetType: AssetType,
  quantity: number,
  prices: Record<AssetType, number>,
  player: PlayerState
): { success: boolean; error?: string; newPlayerState: PlayerState } {
  const currentHoldings = player.portfolio[assetType] || 0;

  // 检查是否有足够持仓
  if (currentHoldings < quantity) {
    return { success: false, error: '持仓不足', newPlayerState: player };
  }

  const price = prices[assetType];
  const totalRevenue = price * quantity;

  // 计算手续费 (1%)
  const fee = Math.round(totalRevenue * 0.01);
  const netRevenue = totalRevenue - fee;

  // 计算滑点
  let slippage = 0;
  const tradeValue = totalRevenue;
  const portfolioValue = Object.entries(player.portfolio).reduce(
    (sum, [type, qty]) => sum + prices[type as AssetType] * qty,
    0
  );
  const totalWealth = player.cash + portfolioValue;

  if (tradeValue > totalWealth * 0.3) {
    slippage = randomBetween(-5, 5);
  }

  // 更新玩家状态
  const newPlayerState = {
    ...player,
    cash: player.cash + netRevenue,
    portfolio: {
      ...player.portfolio,
      [assetType]: currentHoldings - quantity,
    },
    tradeHistory: [
      ...player.tradeHistory,
      {
        id: uuid(),
        day: gameState.currentDay,
        assetType,
        action: 'sell',
        quantity,
        price: price * (1 + slippage / 100),
        total: totalRevenue,
        fee,
      },
    ],
  };

  return { success: true, newPlayerState };
}
```

## 7. 天数推进引擎

```typescript
function advanceDay(gameState: GameState): GameState {
  const { currentDay, prices, player, maxDays } = gameState;

  // 检查是否已到最大天数
  if (currentDay >= maxDays) {
    return {
      ...gameState,
      gamePhase: 'ending',
    };
  }

  // 计算新价格
  const newPrices: Record<AssetType, number> = {};
  Object.values(AssetType).forEach(assetType => {
    newPrices[assetType] = calculateNextPrice(prices[assetType], currentDay, assetType);
  });

  // 记录价格历史
  const newPriceHistory = { ...gameState.priceHistory };
  Object.entries(newPrices).forEach(([assetType, price]) => {
    newPriceHistory[assetType as AssetType] = [
      ...(gameState.priceHistory[assetType as AssetType] || []),
      price,
    ];
  });

  // 计算玩家总资产（新价格下）
  const portfolioValue = Object.entries(player.portfolio).reduce(
    (sum, [type, qty]) => sum + newPrices[type as AssetType] * qty,
    0
  );
  const totalWealth = player.cash + portfolioValue;

  // 选择 NPC 并触发对话
  const randomNPC = getRandomNPC();
  const dialogue = generateDialogue(randomNPC, currentDay, newPrices);

  // 返回新状态
  return {
    ...gameState,
    currentDay: currentDay + 1,
    prices: newPrices,
    priceHistory: newPriceHistory,
    player: {
      ...player,
      totalWealth,
    },
    currentNPC: randomNPC,
    dialogue,
    gamePhase: currentDay + 1 === maxDays ? 'ending' : 'trading',
  };
}
```

```typescript
    mood = 'cautious'; // 温和上涨

    npcId: npc.id,
    text,
    choices,
    mood,
  };
```

## 11. 测试策略

### 11.1 单元测试 (Vitest)

- 价格引擎：测试不同天数、不同资产的价格计算
- 交易引擎：测试买入、卖出、手续费、滑点计算
- 天数推进：测试天数推进后的状态更新
- 对话引擎：测试不同场景下的对话生成

### 11.2 E2E 测试 (Playwright)

- 完整游戏流程测试（开场→交易→结局）
- 不同结局测试（在不同财富区间触发）
- 交易功能测试（买入、卖出、错误处理）
- 天数推进测试（正确推进到 Day 5 并触发结局）
